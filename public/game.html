document.getElementById('generateBtn').addEventListener('click', async () => { try { const response = await fetch('/api/generate-scenario', { method: 'POST' }); const data = await response.json(); // Safely display the generated scenario document.getElementById('scenarioOutput').textContent = JSON.stringify(data, null, 2); } catch (err) { alert(`Error generating scenario: ${err.message}`); console.error('Error:', err); } });
